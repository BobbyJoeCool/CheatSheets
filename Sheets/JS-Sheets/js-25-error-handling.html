Error Handling Examples

// ── try / catch / finally ────────────────────────────────────
try {
  const data = JSON.parse(rawInput);  // may throw SyntaxError
  processData(data);
} catch (err) {
  console.error("Parse failed:", err.message);
} finally {
  cleanup();                  // always runs — success or failure
}

// ── throw ─────────────────────────────────────────────────────
function divide(a, b) {
  if (b === 0) throw new Error("Division by zero");
  return a / b;
}

// Throw any value (not recommended — use Error objects)
throw "something went wrong";   // valid but no stack trace
throw 42;                       // valid but useless
throw new TypeError("Expected string, got number");

// ── Accessing error properties ────────────────────────────────
try {
  null.foo;
} catch (err) {
  console.log(err.name);       // "TypeError"
  console.log(err.message);    // "Cannot read properties of null"
  console.log(err.stack);      // full stack trace string
}

// ── Re-throwing — only catch what you can handle ─────────────
function loadConfig(path) {
  try {
    return JSON.parse(fs.readFileSync(path, "utf8"));
  } catch (err) {
    if (err instanceof SyntaxError) {
      throw new Error(`Invalid JSON in ${path}: ${err.message}`);
    }
    throw err;                  // rethrow unknown errors
  }
}

// ── finally always runs ───────────────────────────────────────
function withConnection(fn) {
  const conn = openConnection();
  try {
    return fn(conn);
  } finally {
    conn.close();               // always close — even if fn throws
  }
}

// ── Checking error type ───────────────────────────────────────
try {
  riskyOperation();
} catch (err) {
  if (err instanceof RangeError) {
    console.warn("Out of range:", err.message);
  } else if (err instanceof TypeError) {
    console.error("Type error:", err.message);
  } else {
    throw err;                  // unknown — rethrow
  }
}

Notes

finally — always executes regardless of whether an error was thrown or caught; use for cleanup (close files, release locks)
throw accepts any value, but always throw Error objects — they carry a stack trace which is essential for debugging
err.message — human-readable description · err.name — the error class name · err.stack — full call stack
Rethrowing — catch only errors you can actually handle; rethrow everything else to avoid swallowing unexpected failures silently
instanceof — use to distinguish error types; more reliable than checking err.name === "TypeError"
finally runs even if there is a return in the try block — the return value is preserved

Common Mistakes

  • Empty catch blocks — silently swallowing errors makes debugging nearly impossible
  • Catching everything and never rethrowing — unexpected errors get hidden instead of surfacing
  • Throwing strings instead of Error objects — no stack trace, no .message property
  • Putting cleanup logic in catch instead of finally — cleanup won't run on the success path

Tips

  • Always throw new Error("message") — the stack trace is the most valuable debugging tool you have
  • Use finally for resource cleanup (DB connections, file handles, locks) — it runs unconditionally
  • Catch specific error types with instanceof and rethrow anything you don't explicitly handle

Summary

  • try wraps risky code · catch handles errors · finally always runs
  • throw new Error("msg") — always throw Error objects; never throw strings
  • err.name · err.message · err.stack — the three key error properties
  • Rethrow what you can't handle — don't swallow unknown errors silently
