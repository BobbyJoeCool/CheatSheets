Partial · Required · Pick · Omit · Readonly · ReturnType · Awaited · Extract

Built-in Utility Types

Partial<T>All props optionalUpdate/patch payloads where any field may be absent
Required<T>All props requiredEnforce all fields after merging defaults

Utility Type Examples

interface User {
  id: number;
  name: string;
  email: string;
  password: string;
  role: "admin" | "user";
}

// ── Partial — all fields optional ────────────────────────────
type UserUpdate = Partial<User>;
// { id?: number; name?: string; email?: string; ... }
function updateUser(id: number, patch: Partial<User>) { /* ... */ }
updateUser(1, { name: "Bob" }); // ✔ only name provided

// ── Required — all fields required ───────────────────────────
type StrictUser = Required<Partial<User>>;  // back to all required

// ── Readonly ──────────────────────────────────────────────────
type FrozenUser = Readonly<User>;
const u: FrozenUser = { id: 1, name: "Alice", email: "", password: "", role: "user" };
u.name = "Bob";  // ✖ Error: readonly

// ── Pick — keep specific fields ───────────────────────────────
type PublicUser = Pick<User, "id" | "name" | "role">;
// { id: number; name: string; role: "admin" | "user" }

// ── Omit — remove specific fields ────────────────────────────
type SafeUser = Omit<User, "password">;
// { id: number; name: string; email: string; role: ... }

// ── Record ────────────────────────────────────────────────────
type RolePermissions = Record<"admin" | "user", string[]>;
const perms: RolePermissions = {
  admin: ["read", "write", "delete"],
  user:  ["read"],
};

// ── Extract / Exclude ─────────────────────────────────────────
type Status = "active" | "inactive" | "pending" | "deleted";
type LiveStatus   = Extract<Status, "active" | "pending">;
// "active" | "pending"
type ArchivedStatus = Exclude<Status, "active" | "pending">;
// "inactive" | "deleted"

// ── NonNullable ───────────────────────────────────────────────
type MaybeString = string | null | undefined;
type DefiniteString = NonNullable<MaybeString>; // string

// ── ReturnType / Parameters ───────────────────────────────────
function getUser(id: number): { name: string; age: number } {
  return { name: "Alice", age: 30 };
}
type UserResult = ReturnType<typeof getUser>;
// { name: string; age: number }

type GetUserParams = Parameters<typeof getUser>;
// [id: number]

// ── Awaited — unwrap Promise ──────────────────────────────────
type AsyncResult = Promise<User[]>;
type Resolved = Awaited<AsyncResult>;  // User[]

async function fetchUsers(): Promise<User[]> { return []; }
type FetchResult = Awaited<ReturnType<typeof fetchUsers>>; // User[]

Notes

Partial<T> is the standard type for update/PATCH payloads — callers can pass only the fields they want to change
Omit<T, "password"> is the idiomatic way to strip sensitive fields when returning user data from an API
ReturnType<typeof fn> — derives the return type automatically; stays in sync when the function signature changes
Awaited<T> — recursively unwraps nested Promises: Awaited<Promise<Promise<string>>> = string
Extract keeps matching union members; Exclude removes them — think "keep" vs "remove"
These are all type-level operations — they produce zero runtime code; used entirely for compile-time type safety

Common Mistakes

  • Confusing Extract and Exclude — Extract keeps matches, Exclude removes them
  • Using Partial for the base type when only some fields should be optional — use Pick + Partial to be precise
  • Forgetting Readonly is shallow — nested objects are still mutable; use recursive mapped types for deep immutability

Tips

  • Compose utility types: Readonly<Pick<User, "id" | "name">> for a readonly public-facing view
  • Use ReturnType<typeof fn> and Awaited<ReturnType<typeof fn>> instead of duplicating return type annotations
  • Use Omit<T, "id" | "createdAt"> to derive a "create" input type from a full entity type automatically

Summary

  • Partial/Required/Readonly — modify all properties at once
  • Pick/Omit — keep or remove specific keys
  • Extract/Exclude/NonNullable — filter union members
  • ReturnType/Parameters/Awaited — infer types from functions and promises
